import com.pega.pegarules.pub.clipboard.ClipboardPage;
import com.pega.pegarules.pub.clipboard.ClipboardPages;
import com.pega.pegarules.pub.context.ThreadContainer;
import com.pega.pegarules.pub.runtime.ParameterPage;

tools oTools = ThreadContainer.get().getTools();

// ── Fixed Parameters ───────────────────────────────────────────
String sRuleType       = "Rule-Obj-Property";
String sAppliesToClass = "PegaFS-Data-Party";

// ── Your 10 RuleNames ──────────────────────────────────────────
String[] ruleNames = {
    "AddressLine1",
    "AddressLine2",
    "AddressLine3",
    "AddressLine4",
    "AddressLine5",
    "AddressLine6",
    "AddressLine7",
    "AddressLine8",
    "AddressLine9",
    "AddressLine10"
};
// ──────────────────────────────────────────────────────────────

// Get work page to store results
ClipboardPage workPage = oTools.getTopLevelPage();
ClipboardPages resultList = workPage.getPageList(".ResultList");

// Loop through each RuleName
for (int i = 0; i < ruleNames.length; i++) {

    String sRuleName = ruleNames[i];

    // Build params
    ParameterPage ppParams = new ParameterPage();
    ppParams.putString("RuleType",       sRuleType);
    ppParams.putString("RuleName",       sRuleName);
    ppParams.putString("AppliesToClass", sAppliesToClass);

    // Call D_ReferencingRules fresh each time (bypasses cache)
    ClipboardPage resultPage = oTools.findDataPage("D_ReferencingRules", ppParams);

    // Append a new row to ResultList
    ClipboardPage newRow = resultList.addPage();
    newRow.putString(".RuleName",       sRuleName);
    newRow.putString(".RuleType",       sRuleType);
    newRow.putString(".AppliesToClass", sAppliesToClass);

    if (resultPage != null) {
        String sPxResultCount = resultPage.getString("pxResultCount");
        newRow.putString(".pxResultCount", sPxResultCount);
        newRow.putString(".Status",        "SUCCESS");
    } else {
        newRow.putString(".pxResultCount", "0");
        newRow.putString(".Status",        "ERROR - null result");
    }

    // Log each result in Pega Tracer
    oTools.getThread().getRequestor().getLog()
          .message("==> [" + (i + 1) + "] RuleName: " + sRuleName
                 + " | pxResultCount: " + newRow.getString(".pxResultCount")
                 + " | Status: "        + newRow.getString(".Status"));
}

// Final summary log
oTools.getThread().getRequestor().getLog()
      .message("===== D_ReferencingRules Batch Complete | Total Processed: " + ruleNames.length + " =====");
