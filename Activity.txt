import fs from 'fs';

// Recursive function to process any value
function processValue(key, value, parentClass, result) {
  if (Array.isArray(value)) {
    // Take only first item
    const firstItem = value[0];
    if (firstItem && typeof firstItem === 'object') {
      const itemClass = firstItem.pxObjectClass || parentClass;
      result.push([key, itemClass]);
      processObject(firstItem, itemClass, result);
    } else {
      // Empty array or primitive array
      result.push([key, parentClass]);
    }
  } else if (value !== null && typeof value === 'object') {
    const objClass = value.pxObjectClass || parentClass;
    result.push([key, objClass]);
    processObject(value, objClass, result);
  } else {
    // Primitive
    result.push([key, parentClass]);
  }
}

function processObject(obj, parentClass, result) {
  const objClass = obj.pxObjectClass || parentClass;
  for (const key of Object.keys(obj)) {
    processValue(key, obj[key], objClass, result);
  }
}

// --- Main ---
const raw = fs.readFileSync('input.json', 'utf-8');
const parsed = JSON.parse(raw);

const result = [];
const rootClass = parsed.pxObjectClass || null;
processObject(parsed, rootClass, result);

console.log('Result:');
result.forEach(([k, v]) => console.log(`[${k}, ${v}]`));

fs.writeFileSync('output.json', JSON.stringify(result, null, 2));
console.log('\noutput.json written!');
